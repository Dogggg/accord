Remarks

Grid Search tries to find the best combination of parameters across a range of possible values that produces the best fit model. If there are two parameters, each with 10 possible values, Grid Search will try an exhaustive evaluation of the model using every combination of points, resulting in 100 model fits.

Examples

How to fit a Kernel Support Vector Machine using Grid Search.

 1// Example binary data
 2double[][] inputs =
 3{
 4    new double[] { -1, -1 },
 5    new double[] { -1,  1 },
 6    new double[] {  1, -1 },
 7    new double[] {  1,  1 }
 8};
 9
10int[] xor = // xor labels
11{
12    -1, 1, 1, -1
13};
14
15// Declare the parameters and ranges to be searched
16GridSearchRange[] ranges = 
17{
18    new GridSearchRange("complexity", new double[] { 0.00000001, 5.20, 0.30, 0.50 } ),
19    new GridSearchRange("degree",     new double[] { 1, 10, 2, 3, 4, 5 } ),
20    new GridSearchRange("constant",   new double[] { 0, 1, 2 } )
21};
22
23
24// Instantiate a new Grid Search algorithm for Kernel Support Vector Machines
25var gridsearch = new GridSearch<KernelSupportVectorMachine>(ranges);
26
27// Set the fitting function for the algorithm
28gridsearch.Fitting = delegate(GridSearchParameterCollection parameters, out double error)
29{
30    // The parameters to be tried will be passed as a function parameter.
31    int degree = (int)parameters["degree"].Value;
32    double constant = parameters["constant"].Value;
33    double complexity = parameters["complexity"].Value;
34
35    // Use the parameters to build the SVM model
36    Polynomial kernel = new Polynomial(degree, constant);
37    KernelSupportVectorMachine ksvm = new KernelSupportVectorMachine(kernel, 2);
38
39    // Create a new learning algorithm for SVMs
40    SequentialMinimalOptimization smo = new SequentialMinimalOptimization(ksvm, inputs, xor);
41    smo.Complexity = complexity;
42
43    // Measure the model performance to return as an out parameter
44    error = smo.Run();
45
46    return ksvm; // Return the current model
47};
48
49
50// Declare some out variables to pass to the grid search algorithm
51GridSearchParameterCollection bestParameters; double minError;
52
53// Compute the grid search to find the best Support Vector Machine
54KernelSupportVectorMachine bestModel = gridsearch.Compute(out bestParameters, out minError);
